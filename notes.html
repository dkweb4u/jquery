            $("#myForm").submit(function (event) {
                event.preventDefault();

                var formData = $(this).serialize();

                $.ajax({
                    url: "process.php", 
                    type: "POST",
                    data: formData,  // or data in object JSON.stringfy(data)
                    success: function(response) {
                        $("#response").html(response); 
                    },
                    error: function(xhr, status, error) {
                        console.error("AJAX Error:", error);
                    }
                });
            });
-> Form to send post Request 

            $.ajax({
                url: "process.php",
                type: "POST",
                data: { name: "John Doe" },
            
                // üöÄ Fired BEFORE the request is sent
                beforeSend: function() {
                    console.log("Before sending AJAX request...");
                },
            
                // ‚úÖ Fired when the request is successful
                success: function(response) {
                    console.log("Success:", response);
                },
            
                // ‚ùå Fired if the request fails
                error: function(xhr, status, error) {
                    console.error("AJAX Error:", error);
                },
            
                // üîÑ Fired when the request is complete (both success & error)
                complete: function() {
                    console.log("AJAX request completed.");
                }
            });
-> added extra function like beforeSend and complete in the ajax request
